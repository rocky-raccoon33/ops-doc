
## 启动

```bash
 docker exec -it obstandalone bash
 obclient -h 127.0.0.1 -P 2881 -u root@sys -p -c -A oceanbase # 密码默认为空 #登陆管理员账号
```

## DDL/DML

```sql
-- 创建资源单元
create resource unit ops_unit max_cpu=1, max_memory='1g', max_iops=128, max_disk_size='5g', max_session_num=64, min_cpu=1, min_memory='1g', min_iops=128; 
 -- 创建资源池
create resource pool ops_pool unit='ops_unit', unit_num=1, zone_list=('zone1');
-- 创建租户
create tenant ops_test resource_pool_list=('ops_pool'); 
```

```sql
-- 登陆租户
obclient -h 127.0.0.1 -P 2881 -u root@ops_test -p -c -A oceanbase 
-- 创建数据库
create database demodb;
-- 创建表
create table a1(id int(4) primary key, name varchar(20));
insert into a1(id, name) values (1,'zhangsan');
select * from a1;
drop table a1;
```

```sql
-- 查看租户信息
select * from __all_tenant;
select * from __all_resource_pool where tenant_id = 1009;
select * from __all_unit_config where unit_config_id in (1061,1062,1063); 
```

```sql
-- 删除租户配置
drop tenant ops_test force; 
drop resource pool ops_pool;
drop resource unit ops_unit;
```
